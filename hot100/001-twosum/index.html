class Solution {  
    public int[] twoSum(int[] nums, int target) {  
        for (int i = 0; i < nums.length; i++) {  
            for (int j = i + 1; j < nums.length; j++) {  
                if (nums[i] + nums[j] == target) {  
                    return new int[]{i, j};  
                }  
            }  
        }  
        return new int[]{};  
    }  
}

class Solution {  
    public int[] twoSum(int[] nums, int target) {
    	// åˆ›å»ºç´¢å¼•æ•°ç»„
        int[] index = new int[nums.length];  
        for (int i = 0; i < nums.length; i++) {  
            index[i] = i;  
        }  
        
        // æ’åº
        quickSort(nums, index, 0, nums.length - 1);  
        
        // åŒæŒ‡é’ˆ
        int leftPointer = 0, rightPointer = nums.length - 1;  
        while (leftPointer < rightPointer) {  
        	// å¦‚æœå’Œç­‰äº target åˆ™è¿”å›
            if (nums[leftPointer] + nums[rightPointer] == target) {  
                int left = index[leftPointer];  
                int right = index[rightPointer];  
                if (left < right) {  
                    return new int[]{left, right};  
                } else {  
                    return new int[]{right, left};  
                }  
            } else if (nums[leftPointer] + nums[rightPointer] < target) {  
                leftPointer++;  
            } else {  
                rightPointer--;  
            }  
        }
        // æ²¡æ‰¾åˆ°è¿”å›ç©ºæ•°ç»„
        return new int[]{};  
    }  
  
    public void quickSort(int[] nums, int[] index, int left, int right) {  
        if (left >= right) return;  
        int i = left, j = right;  
        int pivot = nums[(left + right) >> 1];  
  
        while (i <= j) {  
            while (nums[i] < pivot) i++;  
            while (nums[j] > pivot) j--;  
            if (i <= j) {  
                int temp = nums[i];  
                nums[i] = nums[j];  
                nums[j] = temp;  
                int t = index[i];  
                index[i] = index[j];  
                index[j] = t;  
                i++;  
                j--;  
            }  
        }  
        quickSort(nums, index, left, j);  
        quickSort(nums, index, i, right);  
    }  
}
